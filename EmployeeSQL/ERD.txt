# Simple Conceptual ERD
employees
-
emp_no INT PK
first_name VARCHAR
last_name VARCHAR
birth_date DATE
gender  VARCHAR
hire_date DATE

departments
-
dept_no  VARCHAR PK 
dept_name VARCHAR

titles
-
emo_no INT PK FK - employees.emp_no
title VARCHAR PK
from_date DATE PK
to_date DATE

dept_manager
-
dept_no INT PK FK - departments.dept_no
emp_no INT PK FK - employees.emp_no
from_date DATE
to_date DATE

dept_emp
-
emp_no  INT PK FK - employees.emp_no
dept_no INT PK FK - departments.dept_no
from_date DATE
to_date DATE

salaries
-
emp_no  INT PK FK - employees.emp_no
salary  INT
from_date DATE PK
to_date DATE
